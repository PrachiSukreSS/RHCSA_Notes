####FIREWALL
#firewall is a software that is used to protect the system from unauthorized access and to control 
the incoming and outgoing traffic. It is used to block the unwanted traffic
 and to allow the wanted traffic.
 port no :
 - 80: HTTP
 - 443: HTTPS
 - 22: SSH
 - 25: SMTP
 - 110: POP3    

 practical:
  1. check firewall status:
    >sudo systemctl status firewalld
  2. start firewall:
    >sudo systemctl start firewalld
  3. enable firewall:
    >sudo systemctl enable firewalld

####NFS (Network File System):
#NFS is a protocol that is used to share files and directories over a network. It allows    
port - 2049
package  - nfs-utils
file - /etc/exports

SERVER SIDE:
[root@rhel ~]# history
    1  clear
    2  yum install nfs-utils
    3  mkdir /nfsdata
    4  vi /etc/exports            // WIRTE /nfsdata *(rw,sync)
    5  chmod 777 /nfsdata/
    6  exports -arv
    7  exportfs -arv
    8  systemctl restart nfs-server.service
    9  systemctl enable nfs-server.service
   10  firewall-cmd --permanent --add-service=nfs
   11  firewall-cmd --reload
   12  touch /nfsdata/myfile{1..10}.txt
   13  ls /nfsdata/
   14  ip a
   15  ls /nfsdata/
   16  history 

CLIENT SIDE:

    1  yum install nfs-utils
    2  mkdir /mydata
    3  mount -t nfs 10.0.2.2:/nfsdata /mydata
    4  df -hT
    5  ls /mydata
    6  touch /mydata/linux{1..10}.doc
    7  history
